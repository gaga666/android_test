计算

计算每一个类别的每一个属性的中心趋势，即平均值，在计算概率的时候，使用它作为高斯分布的平均值。

计算每一个类别的每一个属性的变化趋势，即标准差，在计算概率的时候，使用它作为高斯分布的标准差。

计算方差的方法参考计算样本方差时为什么是除以（n-1）？ - 知乎 (zhihu.com)，如果采用除以n的公式计算方差，那么它就不是总体方差的无偏估计了，而是小于总体方差，以下是推导：

三、预测

贝叶斯算法简单来说就是计算数据在每一个类别下的概率，然后选择概率最大的类别作为预测结果

高斯概率密度函数

从之前的步骤中，可以获取到平均值和标准方差

xxxxxxxxxx
def calculate_probability(x, avg, s):
    exponent = math.exp(-(math.pow(x - avg,2) / (2 * math.pow(s,2))))
    return (1/(math.sqrt(2 * math.pi) * s)) * exponent

计算每一个类别概率

仍使用字典存属性值和对应的概率，清晰明了

xxxxxxxxxx
def probabilities(summaries, vector):
    probability = {}
    for classValue, classSummaies in summaries.items():
        probability[classValue] = 1
        for i in range(len(classSummaies)):
            avg, s = classSummaies[i]
            x = vector[i]
            probability[classValue] *= calculate_probability(x,avg,s)
    return probability

import math


import pandas as pd
import xlrd
import csv
import random
def load_csv(filename):
    lines = csv.reader(open(filename, "r"))
    dataset = list(lines)
    for i in range(len(dataset)):
        dataset[i] = [float(x) for x in dataset[i]]
    return dataset


def split_dataset(dataset, split_ratio):
    trainSize = int(len(dataset) * split_ratio)
    trainSet = []
    copy = list(dataset)
    while len(trainSet) < trainSize:
        index = random.randrange(len(copy))
        trainSet.append(copy.pop(index))
    return [trainSet, copy]


def separate_by_class(dataset):
    separate = {}
    for i in range(len(dataset)):
        vector = dataset[i]
        if(vector[-1] not in separate):
            separate[vector[-1]] = []
        separate[vector[-1]].append(vector)
    return separate


def avg(numbers):
    return sum(numbers) / float(len(numbers))


def s(numbers):
    a = avg(numbers)
    var = sum([pow(x-a,2) for x in numbers]) / float(len(numbers)-1)
    return math.sqrt(var)


def combin(dataset):
    com = [(avg(attribute), s(attribute)) for attribute in zip(*dataset)]
    del com[-1]
    return com


def combins(dataset):
    separated = separate_by_class(dataset)
    summaries = {}
    for classValue, instances in separated.items():
        summaries[classValue] = combin(instances)
    return summaries


def calculate_probability(x, avg, s):
    exponent = math.exp(-(math.pow(x - avg,2) / (2 * math.pow(s,2))))
    return (1/(math.sqrt(2 * math.pi) * s)) * exponent


def probabilities(summaries, vector):
    probability = {}
    for classValue, classSummaies in summaries.items():
        probability[classValue] = 1
        for i in range(len(classSummaies)):
            avg, s = classSummaies[i]
            x = vector[i]
            probability[classValue] *= calculate_probability(x,avg,s)
    return probability


def predict(summaries, vector):
    probabilities_ = probabilities(summaries,vector)
    bestLabel, bestProb = None, -1
    for classValue, probability in probabilities_.items():
        if bestLabel is None or probability > bestProb:
            bestProb = probability
            bestLabel = classValue
    return bestLabel


def predicts(summaries, test):
    predictions = []
    for i in range(len(test)):
        result = predict(summaries, test[i])
        predictions.append(result)
    return predictions


def accuracy(test, predictions):
    correct = 0
    for x in range(len(test)):
        if test[x][-1] == predictions[x]:
            correct += 1
    return (correct / float(len(test))) * 100.0


def main():
    filename = '1.csv'
    split_radio = 0.67
    dataset = load_csv(filename)
    train_set, test_set = split_dataset(dataset,split_radio)
    summaries = combins(train_set)
    predictions = predicts(summaries, test_set)
    accuracies = accuracy(test_set, predictions)
    print(('accuracy: {0}%').format(accuracies))
main()
